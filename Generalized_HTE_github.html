Creating some useful functions

cleaR::clear()
{
  
    rmulti = function(prob, n) {
    as.numeric(rMultinom(probs=matrix(prob, 1, length(prob)), m=n))
  }
  
  expit <- function(linearcombination){
    n = (1/(1 + exp(-(linearcombination))))
    return (n)
  }
  
  logit <- function(pr){
    n = log(pr/(1-pr))
    return(n)
  }
  
round_df <- function(df, digits) {
  nums <- vapply(df, is.numeric, FUN.VALUE = logical(1))
  
  df[,nums] <- round(df[,nums], digits = digits)
  (df)
}

diff_fct <- function(x){
  dif <- x - lag(x)
  return(dif)
}


p_load("rlang")
recreate_var <- function(datau, var_name) {
  
  vars <- strsplit(var_name, "__")[[1]]
  
  expr <- rlang::parse_expr(paste(vars, collapse = " * "))
  
  datau %>% mutate("{var_name}" := !!expr)
}

geom_hist_percent <- function(.data, .var) {
  .data %>%
    ggplot(aes(x = !!sym(.var))) +
    geom_histogram(aes(y = after_stat(count/sum(count)))) +
    ylab("Percent") +
    scale_y_continuous(labels = scales::percent) +
    theme_bw()
}


}

data_grf <- tibble(
  age   = rnorm(n, mean=68, sd=10), 
  black = rbinom(n, size=1, prob=0.3), 
  sbp   = rnorm(n, mean=140, sd=15), 
  dbp   = rnorm(n, mean=78, sd=12), 
  scr   = rnorm(n, mean=1.07, sd=0.34), 
  egfr  = rnorm(n, mean=72, sd=20), 
  egfr_ge72 = as.numeric(egfr>=72),
  statin=rbinom(n, size=1, prob=0.43), 
  aspirin=rbinom(n, size=1, prob=0.51), 
  fram  = rnorm(n, mean=25, sd=12), 
  smok  = factor(rmulti(prob=c(0.44, 0.42, 0.14), n))) %>% 
  dummy_cols("smok", remove_selected_columns = F,
             remove_first_dummy = T) %>% 
  mutate(
    tx = rbinom(n, size=1, prob=expit(logit(0.01) + 
                                        log(0.90)*statin+
                                        log(1.01)*sbp+
                                        log(1.01)*dbp+
                                        log(1.01)*age+
                                        log(1.03)*fram+
                                        log(1.1)*black+
                                        log(1.1)*scr+
                                        log(1.1)*smok_3 +
                                        log(1.05)*smok_2)),
    
    y = rbinom(n, size=1, prob=expit(logit(0.01) + 
                                       log(0.038)*tx + 
                                       log(0.592)*egfr_ge72 +
                                       log(0.453)*aspirin +
                                       log(12.00)*tx*egfr_ge72 + 
                                       log(34.70)*tx*aspirin +
                                       log(0.543)*egfr_ge72*aspirin + 
                                       log(0.372)*tx*egfr_ge72*aspirin +
                                       
                                       log(0.90)*statin+
                                       log(1.01)*sbp+
                                       log(1.01)*dbp+
                                       log(1.01)*age+
                                       log(1.03)*fram+
                                       log(1.1)*black+
                                       log(1.2)*scr+
                                       log(1.05)*smok_3 +
                                       log(1.05)*smok_2))) %>% 
  set_variable_labels(egfr    = "Estimated Glomerular Filtration Rate",
                      aspirin = "Aspirin use",

                      dbp     = "Diastolic blood pressure",
                      age     = "Age in years",
                      fram    = "Framingham score risk",
                      black   = "Black race",
                      scr     = "Serum Creatinine",
                      smok_3  = "Current smoker", #smok_1: "Smoking (never)"
